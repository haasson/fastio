    import { test, expect } from '@playwright/test'
    import { createClient } from '@supabase/supabase-js'
    import { fixtures } from './fixtures'

    test.use({ baseURL: 'http://localhost:4710' })

    // Резолвим Supabase URL и service_role из env. Если staging — берём staging,
    // иначе local docker stack (по дефолту localhost:54321). Если service_role
    // не выставлен — skip: без него тест не может прочитать token из tenant_invitations.
    const SB_URL = process.env.SUPABASE_STAGING_URL || 'http://127.0.0.1:54321'
    const SB_SERVICE_ROLE = process.env.SUPABASE_STAGING_SERVICE_ROLE_KEY || process.env.SUPABASE_SERVICE_ROLE_KEY

    test('auth invite staff: admin invites → token from DB → new staff sets password → logs in', async ({ page, context }) => {
      test.skip(!SB_SERVICE_ROLE, 'SUPABASE_SERVICE_ROLE_KEY not set; cannot read invitation token from DB')

      const sb = createClient(SB_URL, SB_SERVICE_ROLE!)
      const inviteEmail = `staff-e2e-${Date.now()}@fastio.test`

      // Step 1: admin logs in (copy admin-login.spec.ts pattern)
      await page.goto('/login')
      await page.getByTestId('admin-login-form').waitFor({ state: 'visible', timeout: 15_000 })
      await page.getByTestId('admin-login-email').locator('input').fill(fixtures.adminEmail)
      await page.getByTestId('admin-login-password').locator('input').fill(fixtures.adminPassword)
      await page.getByTestId('admin-login-submit').click()
      await expect(page.getByTestId('admin-nav')).toBeVisible({ timeout: 20_000 })

      // Step 2: create invite via chosen path (A/B/C from NOTES.md).
      // Если Path C — прямой INSERT через service_role:
      // ... (адаптировать по выбранному пути)

      // Step 3: read token from tenant_invitations
      const { data: inviteRow, error: inviteErr } = await sb
        .from('tenant_invitations')
        .select('id, token, email, expires_at, accepted_at')
        .eq('email', inviteEmail)
        .order('created_at', { ascending: false })
        .limit(1)
        .single()

      expect(inviteErr, 'invitation not created').toBeNull()
      expect(inviteRow?.token, 'invitation token missing').toBeTruthy()
      const inviteToken = inviteRow!.token

      // Step 4: new page → /invite?token=...
      const staffPage = await context.newPage()
      await staffPage.goto(`/invite?token=${inviteToken}`)

      // Step 5: redirect or click "Принять" → /set-password
      // (адаптировать под invite.vue — либо auto-redirect, либо явный submit)
      await staffPage.waitForURL(/\/set-password/, { timeout: 15_000 })

      // Step 6: set password
      const newPassword = 'Staff-E2E-Pass-Test-12345!'
      await staffPage.getByTestId('set-password-form').waitFor({ state: 'visible', timeout: 15_000 })
      await staffPage.getByTestId('set-password-input').locator('input').fill(newPassword)
      // Если есть confirm field — заполнить и его
      const confirmInput = staffPage.getByTestId('set-password-confirm')
      if (await confirmInput.count() > 0) {
        await confirmInput.locator('input').fill(newPassword)
      }
      await staffPage.getByTestId('set-password-submit').click()

      // Step 7: после submit ожидаем admin-nav (auto-login) либо редирект на /login → ручной login
      // Сначала пробуем auto-login:
      const navOrLogin = await Promise.race([
        staffPage.getByTestId('admin-nav').waitFor({ state: 'visible', timeout: 15_000 }).then(() => 'nav' as const),
        staffPage.waitForURL(/\/login/, { timeout: 15_000 }).then(() => 'login' as const),
      ])

      if (navOrLogin === 'login') {
        // если редирект на login — логинимся под новым паролем
        await staffPage.getByTestId('admin-login-form').waitFor({ state: 'visible', timeout: 10_000 })
        await staffPage.getByTestId('admin-login-email').locator('input').fill(inviteEmail)
        await staffPage.getByTestId('admin-login-password').locator('input').fill(newPassword)
        await staffPage.getByTestId('admin-login-submit').click()
        await expect(staffPage.getByTestId('admin-nav')).toBeVisible({ timeout: 20_000 })
      }

      // Cleanup: удалить созданный auth.users (best-effort, чтобы не накапливать)
      try {
        await sb.auth.admin.deleteUser(inviteRow!.id ?? '')
        await sb.from('tenant_invitations').delete().eq('id', inviteRow!.id)
      } catch {
        // не критично — global-setup truncate'нет в следующем прогоне
      }
    })
